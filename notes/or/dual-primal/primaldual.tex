\section{Primal to Dual Conversion}
\subsection{The Primal Problem}
$$
\begin{aligned}
    \max\quad & 5x + 3y + 2z \\
    \text{Subject to} & \\
    & 2x + 3y + z \leq 10 \\
    & 4x + y + 2z \leq 12 \\
    & x, y, z \geq 0 \\
\end{aligned}
$$

\subsection{Standard Form of the Primal Problem}
$$
\begin{aligned}
    \max\quad & 5x + 3y + 2z \\
    \text{Subject to} & \\
    & 2x + 3y + z + s_1 = 10 \\
    & 4x + y + 2z + s_2 = 12 \\
    & x, y, z, s_1, s_2 \geq 0 \\
\end{aligned}
$$

\subsection{The Dual Problem}

$$
\begin{aligned}
    \min\quad & 10y_1 + 12y_2 \\
    \text{Subject to} & \\
    & 2y_1 + 4y_2 \geq 5 \\
    & 3y_1 + 1y_2 \geq 3 \\
    & 1y_1 + 2y_2 \geq 2 \\
    & y_1, y_2 \geq 0 \\
\end{aligned}
$$

\section{Problem 1}
\subsection{The Definition of the Primal Problem}

$$ \begin{aligned}
\max\quad & 3 x + 4 y\\
\text{Subject to} \quad & 3 x + y \leq 24\\
 & x + 4 y \leq 36\\
 & x \geq 0\\
 & y \geq 0\\
\end{aligned} $$

\subsection{Standard Form of the Primal Problem}

$$ \begin{aligned}
\max\quad & 3 x + 4 y\\
\text{Subject to} \quad & 3 x + y + s_1 = 24\\
 & x + 4 y + s_2 = 36\\
 & x \geq 0\\
 & y \geq 0\\
 & s_1 \geq 0\\
 & s_2 \geq 0\\
\end{aligned} 
$$

\subsection{The Dual Problem}

$$ \begin{aligned}
\min\quad & 24 y_1 + 36 y_2\\
\text{Subject to} \quad & 3 y_1 + y_2 \geq 3\\
 & 1 y_1 + 4 y_2 \geq 4\\
 & y_1 \geq 0\\
 & y_2 \geq 0\\
\end{aligned} $$

\section{Problem 2}
\subsection{The Definition of the Primal Problem}
$$ \begin{aligned}
\min\quad & 2 x + 3 y + 4 z\\
\text{Subject to} \quad & x + y + z \geq 12\\

 & x \geq 0\\
 & y \geq 0\\
 & z \geq 0\\
\end{aligned} $$

\subsection{Standard Form of the Primal Problem}
$$ \begin{aligned}
\min\quad & 2 x + 3 y + 4 z\\
\text{Subject to} \quad & x + y + z - s_1 = 12\\
 & 2 x + 3 y + z - s_2 = 24\\
 & x \geq 0\\
 & y \geq 0\\

 & s_2 \geq 0\\
\end{aligned}
